67 int lcd_read_flash(unsigned long addr, unsigned long len, unsigned char* buffer, int attempts)
68 {
69  int res;
70  int remainingAttempts = attempts;
71  if (remainingAttempts <= 0)
72  {
73  remainingAttempts = 1;
74  }
75 
76  while (remainingAttempts > 0)
77  {
78  FLASH_readStr(lcd_writeBuffer, addr, len + 2, 0);
79  res = CalculateCRC16(lcd_writeBuffer, len + 2, 1, 0xFF);
80  if (res == 2)
81  {
82  memcpy(buffer, lcd_writeBuffer, len);
83  return 1;
84  }
85  remainingAttempts--;
86  }
87 
88  return 0;

172 int lcd_check_calibration(unsigned char* calib)
173 {
174  int i;
175  int blank = 1;
176  unsigned int crc;
177 
178  for (i=0; i<SEGMENT_A_LENGTH; i++)
179  {
180  if (calib[i] != 0xFF)
181  {
182  blank = 0;
183  break;
184  }
185  }
186  if (blank)
187  {
188  return 0;
189  }
190 
191  crc = 0;
192  for (i=2; i<SEGMENT_A_LENGTH-1; i += 2)
193  {
194  crc ^= *((unsigned int*)&calib[i]);
195  }
196  return !(crc + *((unsigned int*)&calib[0]));
197 }

741 void lcd_runCfgStateMachine()
742 {
743  int retVal = 0;
744 
745  switch(lcd_cfgState)
746  {
747  case LCDCFG_STATE_ERASE:
748  retVal = FLASH_erase(lcd_writeAddress, lcd_writeLen, lcd_eraseDoneCallback );
749  if(retVal == 0)
750  {
751  lcd_cfgState = LCDCFG_STATE_WAIT_FOR_ERASE;
752  }
753  break;
754  case LCDCFG_STATE_WAIT_FOR_ERASE:
755  break;
756  case LCDCFG_STATE_WRITE:
757  retVal = FLASH_startWrite(lcd_writeAddress, lcd_writeLen );
758  if ( retVal >= 0 )
759  {
760  FLASH_writeStr(lcd_writeBuffer, lcd_writeLen );
761  FLASH_endWriteData();
762  if (lcd_writeType == FLASH_TYPE_CALIB)
763  {
764  lcd_cfgState = LCDCFG_STATE_CFG_IDLE;
765  }
766  else
767  {
768  lcd_cfgState = LCDCFG_STATE_CHECK_CFG;
769  }
770  }
771  break;
772  case LCDCFG_STATE_CHECK_CFG:
773  if(FLASH_getMode() == FLASH_IDLE)
774  {
775  for(retVal=0;retVal<FLASH_BUFFER_SIZE;retVal++)
776  {
777  lcd_writeBuffer[retVal] = 0x00;
778  }
779  FLASH_readStr(lcd_writeBuffer, lcd_writeAddress, lcd_writeLen, 0 );
780  lcd_cfgState = LCDCFG_STATE_SEND_CFG;
781  }
782  break;
783  case LCDCFG_STATE_SEND_CFG:
784  if(FLASH_getMode() == FLASH_IDLE)
785  {
786  retVal = CalculateCRC16(lcd_writeBuffer,(int) lcd_writeLen,1,0xFF);
787  if (retVal != 2)
788  {
789  lcd_writeRemainingAttempts--;
790  if (lcd_writeRemainingAttempts > 0)
791  {
792  // retry write
793  lcd_cfgState = LCDCFG_STATE_ERASE;
794  break;
795  }
796  }
797  switch(lcd_writeType)
798  {
799  case TYPE_TELEMETRY:
800  break;
801  case TYPE_FACTORY:
802  memcpy(factoryConfig_R.bytes, lcd_writeBuffer, sizeof(factoryConfig_t));
803  if (retVal != 2)
804  {
805  lcd_loadFactoryDefaults();
806  telemetry_R.status.config_factoryCRC = 0;
807  }
808  else
809  {
810  telemetry_R.status.config_factoryCRC = 1;
811  }
812  lcd_copyProductInfo();
813  break;
814  case TYPE_USER:
815  memcpy(userConfig_R.bytes, lcd_writeBuffer, sizeof(userConfig_t));
816  if (retVal != 2)
817  {
818  lcd_loadUserDefaults();
819  telemetry_R.status.config_userCRC = 0;
820  }
821  else
822  {
823  telemetry_R.status.config_userCRC = 1;
824  }
825  break;
826  case TYPE_EVENTS:
827  memcpy(eventConfig_R.bytes, lcd_writeBuffer, sizeof(eventConfig_t));
828  if (retVal != 2)
829  {
830  lcd_loadEventsDefaults();
831  telemetry_R.status.config_eventCRC = 0;
832  }
833  else
834  {
835  telemetry_R.status.config_eventCRC = 1;
836  }
837  break;
838  case TYPE_MISC_STATE:
839  memcpy(miscState_R.bytes, lcd_writeBuffer, sizeof(miscState_t));
840  if (retVal != 2)
841  {
842  lcd_loadMiscStateDefaults();
843  telemetry_R.status.config_miscCRC = 0;
844  }
845  else
846  {
847  telemetry_R.status.config_miscCRC = 1;
848  }
849  break;
850  case TYPE_SYS_INFO:
851  break;
852  case FLASH_TYPE_PERSISTENT:
853  if (retVal != 2)
854  {
855  lcd_loadPersistentDefaults();
856  }
857  break;
858  default:
859  break;
860  }
861 
862  lcd_cfgState = LCDCFG_STATE_CFG_IDLE;
863 
864  if (lcd_writeType != FLASH_TYPE_PERSISTENT)
865  {
866  uart_send(lcd_writeType);
867  }
868  }
869  break;
870  case LCDCFG_STATE_CFG_IDLE:
871  break;
872  }
873 }
