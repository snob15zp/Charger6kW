21 /* Telemetry flash storage structure
22  ---------------------------------
23 
24  Each sector consists of:
25  Address | Contents
26  --------|---------
27  000h | Valid sector indicator: will read 0x5A5A if sector contains at least a valid config string
28  | This word should only be written to after the config string for this sector has been verified as correct
29  002h | Block number
30  004h | Timestamp: 8 bytes, milliseconds since unix epoch (1/1/1970 00:00)
31  00Ch | Sample period
32  00Eh | Enable bitfield
33  010h | Start of data
34 
35  The data will be written in 1s complement, and 0xFFFF will be reserved to indicate invalid data, i.e. data that retains its
36  erased value. To convert from 2s complement before writing, subtract 1 from negative values, unless the value is 0x8000 as
37  this would underflow and create a large positive number. The same should be done for 4-byte data.
38 */
39

231 int TELEM_prepareSector()
232 {
233  TelemHeader headerConfirm;
234  unsigned int ii;
235  unsigned int writeSize;
236  unsigned int currentBlock;
237 
238  currentBlock = ( ( telem.currentAddr & ERASE_MASK1 ) - telem.startAddr ) / BLOCK_SIZE;
239 
240  // Fill the telemetry config struct
241  telem.header.blockNum = currentBlock;
242  telem.header.tm = TIME_get();
243  telem.header.period = CFG_remoteCfg.telemSamplePeriod;
244  if ( telem.header.period >= TELEM_PERIOD_END ) telem.header.period = TELEM_PERIOD_END-1;
245  telem.header.enableBitfield = CFG_remoteCfg.telemEnableBitfield;
246 
247  // Determine how many writes will fit in the sector
248  writeSize = 0;
249  if ( telem.header.enableBitfield & TELEM_OUTVOLT_MASK ) writeSize += TELEM_OUTVOLT_SIZE;
250  if ( telem.header.enableBitfield & TELEM_OUTCURR_MASK ) writeSize += TELEM_OUTCURR_SIZE;
251  if ( telem.header.enableBitfield & TELEM_OUTCHARGE_MASK ) writeSize += TELEM_OUTCHARGE_SIZE;
252  if ( telem.header.enableBitfield & TELEM_PVVOLT_MASK ) writeSize += TELEM_PVVOLT_SIZE;
253  if ( telem.header.enableBitfield & TELEM_PVCURR_MASK ) writeSize += TELEM_PVCURR_SIZE;
254  if ( telem.header.enableBitfield & TELEM_PVPOWER_MASK ) writeSize += TELEM_PVPOWER_SIZE;
255  if ( telem.header.enableBitfield & TELEM_PVOCVOLT_MASK ) writeSize += TELEM_PVOCVOLT_SIZE;
256  telem.writesPerSector = ( BLOCK_SIZE - TELEM_DATA_OFFSET ) / writeSize;
257 
258  // Erase this sector
259  if ( FLASH_eraseBlockBusy( telem.currentAddr ) < 0 )
260  {
261  // signal a telemetry error
262  return -1;
263  }
264 
265  // Write the config struct (busy write)
266  for ( ii = 0; ii < sizeof(telem.header); ii++ )
267  {
268  if ( FLASH_writeByteBusy( telem.currentAddr + TELEM_HEADER_OFFSET + ii, *( ((unsigned char *)&(telem.header)) + ii ) ) < 0 ) return -1;
269  }
270 
271  // Read the config string to verify it
272  FLASH_readStr( (unsigned char *)&headerConfirm, telem.currentAddr + TELEM_HEADER_OFFSET, sizeof(headerConfirm), 0 );
273  if ( headerConfirm.blockNum != telem.header.blockNum
274  || headerConfirm.tm != telem.header.tm
275  || headerConfirm.period != telem.header.period
276  || headerConfirm.enableBitfield != telem.header.enableBitfield )
277  {
278  // Config write failed
279  return -1;
280  }
281 
282  // Config string must have verified ok, so write the valid sector marker
283  if ( FLASH_writeByteBusy( telem.currentAddr, ( TELEM_VALID_SECTOR_MARKER >> 8 ) & 0xFF ) < 0 ) return -1;
284  if ( FLASH_writeByteBusy( telem.currentAddr + 1, ( TELEM_VALID_SECTOR_MARKER ) & 0xFF ) < 0 ) return -1;
285 
286  telem.currentAddr += TELEM_DATA_OFFSET;
287 
288  telem.currentTelemBlock = currentBlock;
289 
290  telem.writesLeftThisSector = telem.writesPerSector;
291 
292  return 0;
293 }

315 unsigned int TELEM_getNumBlocksByTime( Time minTm )
316 {
317  int isFound, writeSize;
318  unsigned int marker;
319  TelemHeader headerCheck;
320  unsigned long addr;
321  unsigned int numBlocksByTime;
322 
323  // Go back through the sectors to find the most recent sector before the given time, and return the sector after that
324 
325  numBlocksByTime = 0;
326  addr = telem.currentAddr & ERASE_MASK1;
327  isFound = 0;
328 
329  do
330  {
331  // Read the marker
332  if ( FLASH_readStr( (unsigned char *)&marker, addr, 2, 0 ) != 0 )
333  {
334  return numBlocksByTime;
335  }
336 
337  if ( marker != TELEM_VALID_SECTOR_MARKER )
338  {
339  if ( numBlocksByTime != 0 )
340  {
341  // Found an empty sector that is not the current sector (which could be empty 'cause we haven't written to it yet)
342  isFound = 1;
343  break;
344  }
345  }
346  else
347  {
348  // Read the sector header
349  if ( FLASH_readStr( (unsigned char *)&headerCheck, addr+2, sizeof(headerCheck), 0 ) != 0 )
350  {
351  return numBlocksByTime;
352  }
353 
354  writeSize = 0;
355  if ( headerCheck.enableBitfield & TELEM_OUTVOLT_MASK ) writeSize += TELEM_OUTVOLT_SIZE;
356  if ( headerCheck.enableBitfield & TELEM_OUTCURR_MASK ) writeSize += TELEM_OUTCURR_SIZE;
357  if ( headerCheck.enableBitfield & TELEM_OUTCHARGE_MASK ) writeSize += TELEM_OUTCHARGE_SIZE;
358  if ( headerCheck.enableBitfield & TELEM_PVVOLT_MASK ) writeSize += TELEM_PVVOLT_SIZE;
359  if ( headerCheck.enableBitfield & TELEM_PVCURR_MASK ) writeSize += TELEM_PVCURR_SIZE;
360  if ( headerCheck.enableBitfield & TELEM_PVPOWER_MASK ) writeSize += TELEM_PVPOWER_SIZE;
361  if ( headerCheck.enableBitfield & TELEM_PVOCVOLT_MASK ) writeSize += TELEM_PVOCVOLT_SIZE;
362 
363  if ( ( headerCheck.tm + (Time)(TELEM_PERIOD_MULTIPLIERS[headerCheck.period] * TELEM_BASE_PERIOD_MS) * (Time)(( BLOCK_SIZE - TELEM_DATA_OFFSET )/writeSize) )
364  < minTm )
365  {
366  isFound = 1;
367  //COMMS_sendDebugPacketU64( minTm - ( headerCheck.tm + (Time)(TELEM_PERIOD_MULTIPLIERS[headerCheck.period] * TELEM_BASE_PERIOD_MS) * (Time)(( BLOCK_SIZE - TELEM_DATA_OFFSET )/writeSize) ) );
368  break;
369  }
370  else
371  {
372  // This sector was within the time range
373  numBlocksByTime++;
374  }
375  }
376 
377  addr -= BLOCK_SIZE;
378  if ( addr < telem.startAddr ) addr += telem.memLen;
379  }
380  while ( isFound == 0 && addr != ( telem.currentAddr & ERASE_MASK1 ) );
381 
382  return numBlocksByTime;
