414 void IO_fanSetSpeed()

423  fanFBCount2 = 0;
424 #endif
425 
426  //Check to see if the fan was switched on last time this function was called, if it is running at correct speed.
427  //As this function is called every 2 seconds it should have had plenty of time to start up
428  if (fanDriven){
429 #ifdef ONE_FAN
430  if (curFanFBCount1 < IO_FAN_COUNT_TIMEOUT)
431 #else
432  if (curFanFBCount1 < IO_FAN_COUNT_TIMEOUT || curFanFBCount2 < IO_FAN_COUNT_TIMEOUT)
433 #endif
434  {
435  // Stop the fan and try again in about 30s
436  fanShutdown = 1;
437  SAFETY_fanShutdown();
438  IO_setFanDutyCycle(0);

440  fanRestartTimer = 0;
441  return;
442  }
443  }
444 
445  if (fanShutdown)

447  fanRestartTimer++;
