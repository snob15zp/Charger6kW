390 // This is called at the PWM frequency which is 512us as of Rev 133
391 void CTRL_tick()
392 {
393  // Set if the PWM needs to be stopped this tick
394  int disablePWM = 0;
395 
396  if (IO_getGroundFault()) // RDD !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!! set with delay disablePWM according to GroundFault: J10 on PWRboard
397  {
398  ctrl.pwmGroundFaultShutdown = 1;
399  ctrl.groundFaultTickCount = 0;
400  ctrl.pwmDisabledByOnOff = 1;
401  ctrl.pwmChangeState = -1; // Ignore changes from LCD
402  if (persistentStorage.autoOn)
403  {
404  persistentStorage.autoOn = 0;
405  update_persistent = 1;
406  }
407  disablePWM = 1;
408  }
409  else if (ctrl.pwmGroundFaultShutdown) //RDD what does this affect? On CTRL_getStatus()
410  {
411  // Ground fault pin needs to be up for about 50ms before ground fault is done
412  if (ctrl.groundFaultTickCount > 100)
413  {
414  ctrl.pwmGroundFaultShutdown = 0;
415  }
416  else
417  {
418  ctrl.groundFaultTickCount++;
419  }
420  }
421 
422  if (SAFETY_getStatus() & 0x01)
423  {
424  // Negative PV current
425  disablePWM = 1;
426 
427  // If PWM was on before negative current it will switch on again when current goes positive.
428  // If PWM is turned on by user while the current is negative, the PWM will automatically switch on when current goes positive.
429  // This might display the wrong state if the user tries to turn the output on or off while the current is negative.
430  // This matches the operation of G2.
431  }
432 
433  // Start-up PWM Delay is in place to ensure that the device has correctly
434  // run the safety checks before it does anything else.
435 #ifdef ENABLE_START_UP_PWM_DELAY
436  // Added delay at start-up
437  if(m_unStartUpCount<PWM_START_UP_DELAY_TICKS) { //RDD delays ones at power up.
438  m_unStartUpCount++;
439  return;
440  }
441 #endif
442 
443  if (CFG_toggleSwitchMode)
444  {
445  // Toggle mode - this is not normally used at the moment.
446  if (disablePWM)
447  {
448  // Force a small delay before pwm can be toggled
449  ctrl.onOffDelayTickCount = 0;
450  ctrl.onOffTickCount = 0xFFFF;
451  }
452  else
453  {
454  int pwmWasDisabled = ctrl.pwmDisabledByOnOff;
455  int pwmChangeState = ctrl.pwmChangeState;
456 
457  if (IO_getOnOff())
458  {
459  if (ctrl.onOffTickCount < 0xFFFF)
460  {
461  ctrl.onOffTickCount++;
462  }
463  }
464  else
465  {
466  ctrl.onOffTickCount = 0;
467  }
468 
469  if (ctrl.onOffDelayTickCount < CTRL_ONOFF_DELAY_TICKS)
470  {
471  ctrl.onOffDelayTickCount++;
472  // Must release the button before it will work if pressed before the delay ends
473  ctrl.onOffTickCount = 0xFFFF;
474  }
475  else
476  {
477  if (pwmChangeState != -1)
478  {
479  ctrl.pwmDisabledByOnOff = !pwmChangeState;
480  persistentStorage.autoOn = (uint16_t)((ctrl.pwmDisabledByOnOff == 1) ? 0 : 1); //Invert Logic
481  update_persistent = 1;
482  }
483  else if (ctrl.onOffTickCount == CTRL_ONOFF_HOLD_TICKS)
484  {
485  ctrl.pwmDisabledByOnOff = !ctrl.pwmDisabledByOnOff;
486  persistentStorage.autoOn = (uint16_t)((ctrl.pwmDisabledByOnOff == 1) ? 0 : 1); //Invert Logic
487  update_persistent = 1;
488  }
489 
490  if (ctrl.pwmDisabledByOnOff != pwmWasDisabled || pwmChangeState != -1)
491  {
492  ctrl.onOffDelayTickCount = 0;
493  ctrl.onOffTickCount = 0xFFFF;
494  ctrl.pwmChangeState = -1;
495  }
496  }
497 
498  if (ctrl.pwmDisabledByOnOff)
499  {
500  disablePWM = 1;
501  }
502  }
503  }
504  else
505  {
506  // Remote shutdown or switch mode
507  ctrl.pwmRemoteShutdown = IO_getOnOff();
508 
509  // pwmDisabledByOnOff is always true and autoOn is always false when ground fault shutdown
510  // pwmDisabledByOnOff and autoOn couldn't be changed with a negative PV current for G2 so keep the same
511  if (!disablePWM)
512  {
513  if (ctrl.pwmChangeState != -1)
514  {
515  persistentStorage.autoOn = ctrl.pwmChangeState;
516  ctrl.pwmChangeState = -1;
517  update_persistent = 1;
518  }
519 
520  if (ctrl.pwmRemoteShutdown)
521  {
522  ctrl.pwmDisabledByOnOff = 1;
523  }
524  else
525  {
526  ctrl.pwmDisabledByOnOff = !persistentStorage.autoOn;
527  }
528 
529  if (ctrl.pwmDisabledByOnOff)
530  {
531  disablePWM = 1;
532  }
533  }
534  }
535 
536  if (disablePWM)
537  {
538  // PWM needs to be shutdown
539  ctrl.pwmShutdown = 1;
540 
541  if (!IO_pwmEnabled)
542  {
543  // pvOcVolt is pvVolt when PWM is disabled
544  meas.pvOcVolt.val = meas.pvVolt.val;
545  }
546  IO_disablePwmCtrl();
547  MEAS_setDoUpdate(1);
548  CTRL_enableTurbineLoad = 1;
549  return;
550  }
551  else
552  {
553  if (ctrl.pwmShutdown)
554  {
555  // PWM starting after it has been stopped
556  ctrl.tickCount = 0; //Do a sample. Should give a 100ms debounce period before PWM kicks back in
557  }
558  ctrl.pwmShutdown = 0;
559  }
560 
561  if( meas.caseTempr.val > IQ_cnst(CNTRL_OUTCURR_OVERTEMP_LIMIT_1) && meas.caseTempr.val < IQ_cnst(CNTRL_OUTCURR_OVERTEMP_LIMIT_2) ) // De-rate output current maximum by 20% if inductor is above 90C but below 100C.
562  {
563  unCntrlOutCurrLimit=IQ_cnst( (CNTRL_OUTCURR_LIMIT*CNTRL_OUTCURR_OVERTEMP_REDUC_1) / MEAS_OUTCURR_BASE );
564  unCntrlOutCurrSwOffPoint=IQ_cnst( ((CNTRL_OUTCURR_LIMIT*CNTRL_OUTCURR_OVERTEMP_REDUC_1)*CNTRL_OUTCURR_HYST_PERCENT/100.) / MEAS_OUTCURR_BASE );
565  }
566  else if( meas.caseTempr.val > IQ_cnst(CNTRL_OUTCURR_OVERTEMP_LIMIT_2)) // De-rate output current maximum by 30% if inductor is above 100C (Additional Temperature Failsafe before 120C Shutdown).
567  {
568  unCntrlOutCurrLimit=IQ_cnst( (CNTRL_OUTCURR_LIMIT*CNTRL_OUTCURR_OVERTEMP_REDUC_2) / MEAS_OUTCURR_BASE );
569  unCntrlOutCurrSwOffPoint=IQ_cnst( ((CNTRL_OUTCURR_LIMIT*CNTRL_OUTCURR_OVERTEMP_REDUC_2)*CNTRL_OUTCURR_HYST_PERCENT/100.) / MEAS_OUTCURR_BASE );
570  }
571  else

850 void CTRL_checkBulkFloat()
851 {
852  Iq outVoltSetpointIq;
853  //if ( ctrl.tickCount < CTRL_OUT_VOLT_BLANK_TICKS ) return;
854 
855  //IO_flset( ctrl.flsetFloat );
856 
857  if ( IO_getIsSlave() )

860  }
861 
862  if ( ctrl.setpointIsBulk ) outVoltSetpointIq = ctrl.bulkVolt;
863  else outVoltSetpointIq = ctrl.floatVolt;
864 
865 // COMMS_sendDebugPacket( meas.outVolt.val, outVoltSetpointIq, ctrl.bulkTime, ctrl.timeAtSetpoint );
866 
867  if ( IQ_abs( meas.outVolt.val - outVoltSetpointIq ) < IQ_cnst(1.4/MEAS_OUTVOLT_BASE) )
868  {
869  // We're at the output voltage setpoint
870  ctrl.mode = CTRL_MODE_OUT_REG;
871  if ( !IO_getIsSlave() )
872  {
873  if ( ctrl.setpointIsBulk )
874  {
875  // Increment the bulk timer
876  ctrl.timeAtSetpoint += CTRL_SLOW_PERIOD_MS;
877  if ( ctrl.timeAtSetpoint >= ctrl.bulkTime )
878  {
879  //IO_flset(ctrl.flsetFloat);
880  //PWM_setFlTrim( ctrl.fltrimDutyFloat );

887  ctrl.timeAtSetpoint = 0;
888  }
889  }
890  }
891  else
892  {
893  if ( meas.outVolt.val < outVoltSetpointIq )
894  {
895  // We're below the bulk voltage setpoint, so must be in MPPT regulation mode
896  ctrl.mode = CTRL_MODE_MPPT_REG;
897  }
898  else
899  {
900  // We're above the bulk voltage setpoint, so must be limiting on maximum input voltage
901  ctrl.mode = CTRL_MODE_VIN_LIM;

904  if (!ctrl.setpointIsBulk || ctrl.notBulkTime >= CTRL_BULK_HYSTERISIS_MS)

907  }
908  else
909  {
910  ctrl.notBulkTime += CTRL_SLOW_PERIOD_MS;
911  }
912  }
913 
914  if ( !IO_getIsSlave() )
915  {
916  // Regulate bulk if we drop below the bulk reset voltage
917  if ( meas.outVolt.val <= ctrl.bulkResetVolt )
918  {
919  //IO_flset(ctrl.flsetBulk);
920  //PWM_setFlTrim( ctrl.fltrimDutyBulk );
921  ctrl.setpointIsBulk = 1;
