111 void uart_init( void )
112 {
113  RCC->APB2ENR |= RCC_APB2ENR_USART1EN; //USART1 clock enable
114  RCC->AHBENR |= RCC_AHBENR_GPIOCEN; //GPIOC clock enable
115 
116  GPIOC->MODER &= ~(GPIO_MODER_MODER4_0 | GPIO_MODER_MODER4_1);
117  GPIOC->MODER &= ~(GPIO_MODER_MODER5_0 | GPIO_MODER_MODER5_1);
118  GPIOC->MODER |= GPIO_MODER_MODER4_1 | GPIO_MODER_MODER5_1;
119  GPIOC->OTYPER &= ~GPIO_OTYPER_OT_4;
120 // GPIOC->PUPDR &= ~(GPIO_PUPDR_PUPDR5_0 | GPIO_PUPDR_PUPDR5_1);
121 // GPIOC->PUPDR |= GPIO_PUPDR_PUPDR5_1;
122 
123  GPIOC->AFR[0] |= (ALT_FUNC_USART<<GPIO_AFRL_AFRL4_Pos) | (ALT_FUNC_USART<<GPIO_AFRL_AFRL5_Pos);
124 
125  USART1->BRR=USART1_FCLK/USART1_BAUD;
126  USART1->CR1 = USART_CR1_UE;
127  USART1->CR1 |= USART_CR1_TE;
128  USART1->CR1 |= USART_CR1_RE;
129  USART1->CR1 |= USART_CR1_RXNEIE;
130  NVIC_EnableIRQ(USART1_IRQn);
131 }

188 /*
189  * Process the recevied data.
190  * If the received data is a request then send the requested packet otherwise write the data to flash then
191  * send it back to the sender so it can validate that the correct data was written.
192  */
193 void uart_receive(void)
194 {
195  unsigned char i;
196  unsigned char k;
197  unsigned char crc_start;
198  packetIdentifier_t packetID;
199  unsigned char *bytes = 0;
200  unsigned char structSize = 0;
201  unsigned char version = 0;
202 
203  if(!sent_startup)
204  {
205  sent_startup = 1;
206  uart_send_startup();
207  return;
208  }
209 
210  if(uart_state == UART_STATE_RECEIVED_PACKET)
211  {
212  crc_start = rx_count - FOOTER_LENGTH;
213  k = crc_start;
214  // Un-stuff crc bytes
215  if(rx_buffer[k] == DLE)
216  {
217  k++;
218  }
219  k++;
220  if(rx_buffer[k] == DLE)
221  {
222  k++;
223  }
224  rx_buffer[crc_start+1] = rx_buffer[k];
225  if(CalculateCRC16(rx_buffer,crc_start+2,1,0x00) == 2)
226  {
227  // The CRC was correct so process the packet
228  k = 4;
229  if(rx_buffer[k] == DLE)
230  {
231  k++;
232  }
233  packetID.byte = rx_buffer[k++];
234  switch(packetID.type)
235  {
236  case TYPE_TELEMETRY:
237  break;
238  case TYPE_FACTORY:
239  bytes = factoryConfig_W.bytes;
240  structSize = sizeof(factoryConfig_t);
241  version = VERSION_FACTORY;
242  break;
243  case TYPE_USER:
244  bytes = userConfig_W.bytes;
245  structSize = sizeof(userConfig_t);
246  version = VERSION_USER;
247  break;
248  case TYPE_EVENTS:
249  bytes = eventConfig_W.bytes;
250  structSize = sizeof(eventConfig_t);
251  version = VERSION_EVENTS;
252  break;
253  case TYPE_SYS_INFO:
254  break;
255  case TYPE_COMMAND:
256  bytes = command_W.bytes;
257  structSize = sizeof(command_t);
258  version = VERSION_COMMAND;
259  break;
260  case TYPE_PASSWORD:
261  break;
262  case TYPE_SET_TIME:
263  bytes = setTime_W.bytes;
264  structSize = sizeof(setTime_t);
265  version = VERSION_SET_TIME;
266  break;
267  case TYPE_MISC_STATE:
268  bytes = miscState_W.bytes;
269  structSize = sizeof(miscState_t);
270  version = VERSION_MISC_STATE;
271  break;
272  default:
273  // Unknown packet type
274  uart_send_error(ERROR_PACKET_TYPE);
275  return;
276  }
277 
278  if(packetID.request)
279  {
280  if (packetID.type == TYPE_TELEMETRY)
281  {
282  // Reload telemetry data each time.
283  // The other packets are only loaded at startup or when the flash is modified.
284  lcd_loadTelemetry();
285  }
286  else if (packetID.type == TYPE_PASSWORD)
287  {
288  loadPassword();
289  }
290  if (packetID.type != TYPE_COMMAND && packetID.type != TYPE_SET_TIME)
291  {
292  uart_send(packetID.type);
293  return;
294  }
295  }
296  else if(bytes)
297  {
298  if(packetID.version != version)
299  {
300  uart_send_error(ERROR_VERSION);
301  return;
302  }
303 
304  i = 0;
305  // Un-stuff received data and load it into the appropriate struct
306  while(k < crc_start)
307  {
308  if(rx_buffer[k] == DLE)
309  {
310  k++;
311  }
312  if (i >= structSize)
313  {
314  // Too much data
315  i = -1;
316  break;
317  }
318  if (k >= crc_start)
319  {
320  // Bad stuffing - DLE at the end of the data buffer
321  i = -1;
322  break;
323  }
324  bytes[i++] = rx_buffer[k++];
325  }
326  if(i == structSize)
327  {
328  if(packetID.type == TYPE_COMMAND)
329  {
330  if(command_W.commandCode == COMMAND_RESET)
331  {
332  // Force watchdog reset
335  return;
336  }
337  else if (command_W.commandCode == COMMAND_ENABLE_OUTPUT)
338  {
339  CTRL_enableOutput(command_W.arg ? 1 : 0);
340  uart_send_response(COMMAND_ENABLE_OUTPUT, command_W.arg ? 1 : 0);
341  return;
342  }
343  }
344  else if(packetID.type == TYPE_SET_TIME)
345  {
346  TIME_set(setTime_W.time);
347  }
348  else
349  {
350  // Successfully received the packet so verify and start writing to flash
351  if(lcd_queueWrite(packetID.type))
352  {
353  return;
354  }
355  }
356  }
357  }
358  }
359  uart_state = UART_STATE_IDLE;
360  }
361  else
362  {
363  if (uart_state == UART_STATE_IDLE)
364  {
365  lcd_checkPersistentUpdate();
366  }
367  }

370 int uart_send(int type)
371 {
372  unsigned char *bytes;
373  unsigned char structSize;
374  unsigned char version;
375  int i;
376  int k;
377  unsigned char c;
378  unsigned char d;
379  packetIdentifier_t packetID;
380 
381  uart_state = UART_STATE_SENDING_PACKET;
382 
383  switch(type)
384  {
385  case TYPE_TELEMETRY:
386  bytes = telemetry_R.bytes;
387  structSize = sizeof(telemetry_t);
388  version = VERSION_TELEMETRY;
389  break;
390  case TYPE_FACTORY:
391  bytes = factoryConfig_R.bytes;
392  structSize = sizeof(factoryConfig_t);
393  version = VERSION_FACTORY;
394  break;
395  case TYPE_USER:
396  bytes = userConfig_R.bytes;
397  structSize = sizeof(userConfig_t);
398  version = VERSION_USER;
399  break;
400  case TYPE_EVENTS:
401  bytes = eventConfig_R.bytes;
402  structSize = sizeof(eventConfig_t);
403  version = VERSION_EVENTS;
404  break;
405  case TYPE_SYS_INFO:
406  bytes = sysInfo_R.bytes;
407  structSize = sizeof(sysInfo_t);
408  version = VERSION_SYS_INFO;
409  break;
410  case TYPE_COMMAND:
411  bytes = command_R.bytes;
412  structSize = sizeof(command_t);
413  version = VERSION_COMMAND;
414  break;
415  case TYPE_PASSWORD:
416  bytes = password_R.bytes;
417  structSize = sizeof(password_t);
418  version = VERSION_PASSWORD;
419  break;
420  case TYPE_MISC_STATE:
421  bytes = miscState_R.bytes;
422  structSize = sizeof(miscState_t);
423  version = VERSION_MISC_STATE;
424  break;
425  default:
426  uart_state = UART_STATE_IDLE;
427  return 0;
428  }
429 
430  packetID.type = type;
431  packetID.request = 0;
432  packetID.version = version;
433 
434  tx_buffer[0] = DLE;
435  tx_buffer[1] = STX;
436 
437  // Length is written in index 2 and 3 later
438 
439  k = 4;
440  if (packetID.byte == DLE)
441  {
442  tx_buffer[k++] = DLE;
443  }
444  tx_buffer[k++] = packetID.byte;
445 
446  // Copy data from the struct
447  for (i=0; i<structSize; i++)
448  {
449  if (bytes[i] == DLE)
450  {
451  if(k >= PACKET_LENGTH - FOOTER_LENGTH)
452  {
453  // Error: Too much data
454  return 0;
455  }
456  tx_buffer[k++] = DLE;
457  }
458  if(k >= PACKET_LENGTH - FOOTER_LENGTH)
459  {
460  // Error: Too much data
461  return 0;
462  }
463  tx_buffer[k++] = bytes[i];
464  }
465 
466  tx_buf_len = k+FOOTER_LENGTH;
467  // Write total packet length
468  if (tx_buf_len == DLE)
469  {
470  tx_buffer[2] = DLE;
471  tx_buffer[3] = DLE;
472  }
473  else
474  {
475  tx_buffer[2] = tx_buf_len;
476  tx_buffer[3] = 0;
477  }
478 
479  CalculateCRC16(tx_buffer,k+2,0,0x00);
480  c = tx_buffer[k];
481  d = tx_buffer[k+1];
482  if (c == DLE)
483  {
484  tx_buffer[k++] = DLE;
485  }
486  tx_buffer[k++] = c;
487  if (d == DLE)
488  {
489  tx_buffer[k++] = DLE;
490  }
491  tx_buffer[k++] = d;
492  for(; k<tx_buf_len-2; k++)
493  {
494  tx_buffer[k] = 0;
495  }
496 
497  tx_buffer[k++] = DLE;
498  tx_buffer[k++] = ETX;
499 
500  uart_tx();
501 
502  return 1;
503 }

533 void USART1_IRQHandler(void)
534 {
535  unsigned char rx;
536  Time time;
537 
538  //IRQ transmit complete
539  if(USART1->ISR & USART_ISR_TC){
541  USART1->CR1 &= ~USART_CR1_TCIE; //tx interrupt disable
543  if (uart_state != UART_STATE_SENDING_PACKET)
544  {
546  USART1->CR1 &= ~USART_CR1_TCIE; //tx interrupt disable
547  }
548  else if (tx_count < tx_buf_len)
549  {
551  USART1->CR1 |= USART_CR1_TCIE; //tx interrupt enable
552  USART1->TDR = tx_buffer[tx_count++];
554  }
555  else
556  {
557  // Finished sending so start listening for new requests.
558  uart_state = UART_STATE_IDLE;
560  USART1->CR1 &= ~USART_CR1_TCIE; //tx interrupt disable
561  }
562  }
563 
564  //IRQ receive complete
565  if(USART1->ISR & USART_ISR_RXNE){
567  USART1->CR1 &= ~USART_CR1_RXNEIE; //rx interrupt disable
570  rx = (unsigned char)USART1->RDR;
571  if((uart_state != UART_STATE_RECEIVED_PACKET) && (uart_state != UART_STATE_SENDING_PACKET))
572  {
573  // Reset the state if the last byte was received more than 1s ago.
574  time = TIME_get();
575  if(time > prev_time + 1000 || time < prev_time)
576  {
577  uart_state = UART_STATE_IDLE;
578  prev_rx = 0xFF;
579  }
580  prev_time = time;
581 
582  if((rx == STX) && (prev_rx == DLE))
583  {
584  // A packet has started
585  rx_buffer[0] = DLE;
586  rx_buffer[1] = STX;
587  rx_count = 2;
588  expected_rx_count = 0;
589  uart_state = UART_STATE_RECEIVING_PACKET;
590  }
591  else if(uart_state == UART_STATE_RECEIVING_PACKET)
592  {
593  rx_buffer[rx_count++] = rx;
594  if(rx_count <= 4)
595  {
596  if (((rx_count == 3) && (rx != DLE)) || ((rx_count == 4) && (prev_rx == DLE)))
597  {
598  expected_rx_count = rx;
599  }
600  }
601  else if(rx_count <= expected_rx_count)
602  {
603  if((prev_rx == DLE) && (rx == ETX))
604  {
605  if (rx_count == expected_rx_count)
606  {
607  uart_state = UART_STATE_RECEIVED_PACKET;
608  }
609  else
610  {
611  // Length doesn't match
612  uart_state = UART_STATE_IDLE;
613  }
614  }
615  }
616  else
617  {
618  // Bad packet - reset state
619  uart_state = UART_STATE_IDLE;
620  }
621  }
622  else if ((rx == DLE) && (prev_rx == DLE))
623  {
624  rx = 0xFF;
625  }
626  }
627 
628  prev_rx = rx;
630  USART1->CR1 |= USART_CR1_RXNEIE;
631  }
