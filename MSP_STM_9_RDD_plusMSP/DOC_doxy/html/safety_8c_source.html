106 void SAFETY_tick()
107 {
108 
109  if ( meas.pvCurr.valPreFilter < IQ_cnst(-2.0/MEAS_PVCURR_BASE) ) // PV Negative Current Shutdown
110  {
111  safety.pvCurrNegTickCnt++;
112  if ( safety.pvCurrNegTickCnt >= PVCURR_NEG_MIN_TICKS )
113  {
114  safety.sdBits |= ( 1 << SAFETY_SD_BIT_PVCURR_NEG );
115  }
116  }
117  else
118  {
119  safety.pvCurrNegTickCnt = 0;
120  }
121 
122  if ( ( IQ_abs(meas.pvCurr.valPreFilter) > SAFETY_PulseCurrLimit ) ) // PV Pulse Current Shutdown
123  {
124  safety.sdBits |= ( 1 << SAFETY_SD_BIT_PVCURR_POS );
125  }
126 
127  if ( meas.pvVolt.val > SAFETY_PVVoltLimit ) // PV Voltage Shutdown
128  {
129  safety.sdBits |= ( 1 << SAFETY_SD_BIT_PVVOLT );
130  }
131 
132  if ( meas.outCurr.val > SAFETY_OutCurrCrit ) // Immediate Output Overcurrent Shutdown
133  {
134  safety.sdBits |= ( 1 << SAFETY_SD_BIT_OUTCURR_POS );
135  }
136 
137  if ( meas.outCurr.val > SAFETY_OutCurrLimit ) // Backup Delayed Overcurrent Shutdown (for Current Limiting)
138  {
139  if (safety.outCurrDelayTickCnt >= OUTCURR_DELAY_TICKS )
140  {
141  safety.sdBits |= ( 1 << SAFETY_SD_BIT_OUTCURR_POS );
142  }
143  else
144  {
145  safety.outCurrDelayTickCnt++;
146  }
147  }
148  else
149  {
150  safety.outCurrDelayTickCnt = 0;
151  }
152 
153  if ( meas.outVolt.val > SAFETY_OutVoltLimit ) // Output Overvoltage Shutdown
154  {
155  if ( safety.outVoltOCDelayTickCnt >= OUTVOLT_OC_DELAY_TICKS )
156  {
157  safety.sdBits |= ( 1 << SAFETY_SD_BIT_OUTVOLT );
158  }
159  else
160  {
161  safety.outVoltOCDelayTickCnt++; // Delay Out overvolt check for MPPT Track.
162  }
163  }
164  else
165  {
166  safety.outVoltOCDelayTickCnt = 0;
167  }
168 
169  if ( meas.pvVolt.valReal <= (meas.outVolt.valReal + SAFETY_IO_VOLTAGE_THRESHOLD)) // Protects against PV Breaker Disconnect/Reconnect Hard Starts & Reverse PV & Output being enabled before PV is connected.
170  {
171  safety.sdBits |= ( 1 << SAFETY_SD_BIT_PANEL_MISSING );
172  }
173 
174  if ( meas.caseTempr.val > IQ_cnst(SAFETY_CASETMP_LIMIT) )
175  {
176  safety.sdBits |= ( 1 << SAFETY_SD_BIT_CASETMP );
177  }
178  else if ( meas.caseTempr.val < IQ_cnst(SAFETY_CASETMP_RESET) )
179  {
180  if(safety.sdBits & ( 1 << SAFETY_SD_BIT_CASETMP ) )
181  {
182  safety.sdBits &= ~( 1 << SAFETY_SD_BIT_CASETMP );
183  }
184  }
185 
186  // If any shutdown bits, go into shutdown mode
187  // WORKAROUND: fan fault
188  //if ( safety.sdBits != 0 )
189  if ( (safety.sdBits & (~(1 << SAFETY_SD_BIT_FAN))) != 0 )
190  {
192  }
193 }

195 void SAFETY_monitor()
196 {
197 
198  // Delayed retry for PV_OVERVOLT
199  // The counter will be reset anytime there is a shutdown event.
200  // Counter will remain at zero if the PV_OVERVOLT condition persists.
201 
202 if (safety.sdBits & ( 1 << SAFETY_SD_BIT_PVVOLT ) || safety.sdBits & ( 1 << SAFETY_SD_BIT_OUTCURR_POS ))
203  {
204  if ( meas.pvVolt.val < SAFETY_PVVoltLimit ) // Check if PV Voltage is Safe
205  {
206  if (safety.delayedRetryTickCnt == SAFETY_DELAYED_RETRY_TICKS)
207  {
208  safety.sdBits &= ~( 1 << SAFETY_SD_BIT_PVVOLT ); // Clear the bits after the retry period has elapsed.
209  safety.sdBits &= ~( 1 << SAFETY_SD_BIT_OUTCURR_POS );
210  safety.delayedRetryTickCnt = 0;
211  }
212  else
213  {
214  safety.delayedRetryTickCnt++;
215  }
216  }
217  else
218  {
219  safety.delayedRetryTickCnt = 0;
220  }
221  }
222 
223 if (safety.sdBits & ( 1 << SAFETY_SD_BIT_OUTVOLT ))
224  {
225  if ( ( meas.outVolt.val >= SAFETY_OutVoltRstLow ) && ( meas.outVolt.val <= SAFETY_OutVoltRstHigh ) ) // Check for a normal battery voltage reading before re-activating the output.
226  {
227 
228  if ( safety.outVoltRstDelayTickCnt >= OUTVOLT_RST_DELAY_TICKS )
229  {
230  safety.sdBits &= ~( 1 << SAFETY_SD_BIT_OUTVOLT );
231  safety.outVoltRstDelayTickCnt = 0;
232  }
233  else
234  {
235  safety.outVoltRstDelayTickCnt++;
236  }
237  }
238  else
239  {
240  safety.outVoltRstDelayTickCnt = 0;
241  }
242  }
243 
244 if (safety.sdBits & ( 1 << SAFETY_SD_BIT_PANEL_MISSING ))
245  {
246 
247  if ( meas.pvVolt.valReal >= (meas.outVolt.valReal + SAFETY_IO_VOLTAGE_THRESHOLD) )
248  {
249  if ( safety.lowPVShutdownInt == 0 ) // If this is the first time this has triggered since startup, clears alarm in 5 seconds.
250  {
251  if ( safety.PVVoltLowSrtupDelayTickCnt >= PVVOLTLOW_SRTUP_DELAY_TICKS )
252  {
253  safety.sdBits &= ~( 1 << SAFETY_SD_BIT_PANEL_MISSING );
254  safety.PVVoltLowSrtupDelayTickCnt = 0;
255 
256  safety.lowPVShutdownInt = 1;
257  }
258  else
259  {
260  safety.PVVoltLowSrtupDelayTickCnt++;
261  }
262  }
263  else // If not startup sequence/first time turning on unit/pv, delay alarm clear for 20 minutes to prevent end of day PV voltage bouncing during sunset.
264  {
265  if ( safety.PVVoltLowRstDelayTickCnt >= PVVOLTLOW_RST_DELAY_TICKS )
266  {
267  safety.sdBits &= ~( 1 << SAFETY_SD_BIT_PANEL_MISSING );
268  safety.PVVoltLowRstDelayTickCnt = 0;
269  }
270  else
271  {
272  safety.PVVoltLowRstDelayTickCnt++;
273  }
274  }
275  }
276  else
277  {
278  safety.PVVoltLowSrtupDelayTickCnt = 0;
279  safety.PVVoltLowRstDelayTickCnt = 0;
280  }
281 
282  }
283 }
