139 // Erase all sectors touched by the given address range.
140 // This call starts erasing; FLASH_tick() continues it.
141 int FLASH_erase( unsigned long dstAddr, unsigned long len, void (*pCall)(int) )
142 {
143  // Return if flash activity in progress already
144  if ( flash.mode ) return -1;
145 
146  flash.mode = FLASH_ERASE_ACTIVE;
147 
148  // Round address down to start of smallest erasable block containing start address
149  flash.currentAddr = dstAddr & ERASE_MASK1;
150  // Round length up to end of smallest erasable block containing end address
151  flash.lenRem = ( ( len + ( dstAddr - flash.currentAddr ) - 1 ) & ERASE_MASK1 ) + ERASE_SIZE1;
152 
153  flash.pCallback = pCall;
154 
155  return 0;
156 }

172 // Copy data to be written into circular write buffer, and start write process
173 // Return amount of data added to buffer
174 int FLASH_writeStr( unsigned char * pSrc, unsigned long len )
175 {
176  unsigned int ii, bytesToWriteThis;
177 
178  if ( flash.mode == FLASH_WRITE_ACTIVE )
179  {
180  bytesToWriteThis = ( flash.lenRem < len ) ? flash.lenRem : len;
181 
182  for ( ii = 0; ii < bytesToWriteThis; ii++ )
183  {
184  if ( UTIL_putToCircBuffer( &flash.buffer, pSrc[ii] ) < 0 )
185  {
186  break;
187  }
188  }
189 
190  flash.lenRem -= ii;
191 
192  return ii;
193  }
194 
195  return -1;
196 }
