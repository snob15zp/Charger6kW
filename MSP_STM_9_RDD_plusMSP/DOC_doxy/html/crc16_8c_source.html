40 /********************************************************************
41  * Name : CRC16_Update4Bits
42  * Description : A helper function to the main CRC calculation.
43  * Process 4 bits of the message to update the CRC Value.
44  * Inputs : val - data whose CRC is to be calculated
45  * Note that the data must be in the low nibble of val.
46  * Outputs : None
47  * Return Value: None
48  * Side Effects: Global "CRC16_High","CRC16_Low" are modified. (hardcoded)
49  * History : 12-Feb-2008 (PD)
50  * Function header and comments.
51  ********************************************************************/
52 void CRC16_Update4Bits( unsigned char val )
53 {
54  unsigned char t;
55 
56  // Step one, extract the Most significant 4 bits of the CRC register
57  t = CRC16_High >> 4;
58 
59  // XOR in the Message Data into the extracted bits
60  t = t ^ val;
61 
62  // Shift the CRC Register left 4 bits
63  CRC16_High = (CRC16_High << 4) | (CRC16_Low >> 4);
64  CRC16_Low = CRC16_Low << 4;
65 
66  // Do the table lookups and XOR the result into the CRC Tables
67  CRC16_High = CRC16_High ^ CRC16_LookupHigh[t];
68  CRC16_Low = CRC16_Low ^ CRC16_LookupLow[t];
69 }
