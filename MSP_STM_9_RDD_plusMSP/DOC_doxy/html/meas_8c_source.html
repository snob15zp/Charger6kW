309 Iq MEAS_filter( Iq valNow, Iq valIn )
310 {
311  valIn -= valNow;
312  if (IQ_abs(valIn) <= ((int)(1.0/(float)CORNERFREQ)+1))
313  {
314  if (valIn < 0) valNow--;
315  else if (valIn > 0) valNow++;
316  }
317  else valNow += IQ_mpy( IQ_cnst(CORNERFREQ), valIn );
318  return valNow;
